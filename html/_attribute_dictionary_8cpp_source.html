90 std::vector<std::string> AttributeDictionary::get_keys() const {
91  if (!(*this)) {
92  throw std::logic_error("CW::AttributeDictionary::get_keys(): AttributeDictionary is null");
93  }
94  size_t num_keys = 0;
95  SUResult res = SUAttributeDictionaryGetNumKeys(m_dict, &num_keys);
96  assert(res == SU_ERROR_NONE);
97  SUStringRef* keys_ref = new SUStringRef[num_keys];
98  for (size_t i=0; i < num_keys; i++) {
99  keys_ref[i] = SU_INVALID;
100  SUStringCreate(&keys_ref[i]);
101  }
102  SUAttributeDictionaryGetKeys(m_dict, num_keys, &keys_ref[0], &num_keys);
103  std::vector<std::string> keys;
104  keys.reserve(num_keys);
105  for (size_t i=0; i < num_keys; i++) {
106  keys.push_back(String(keys_ref[i]));
107  }
108  delete[] keys_ref;
109  return keys;
110 }
