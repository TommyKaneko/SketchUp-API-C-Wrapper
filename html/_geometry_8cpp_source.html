47 Radians::Radians(const double &rhs):
48  m_val(rhs)
49 {
50  if(rhs > (2*PI)) {
51  int numPi = rhs / (2*PI); // Get number of times 2pi fits in to rhs
52  m_val = rhs - (numPi*2*PI);
53  }
54  else if(rhs < 0.0) {
55  int numPi = (rhs / (2*PI)) - 1; // Get number of times 2pi fits in to rhs
56  m_val = rhs - (numPi*2*PI);
57  }
58 }

283 Vector3D::Colinearity Vector3D::colinear(const Vector3D& vector_b) const {
284  if (this->length() < EPSILON || vector_b.length() < EPSILON) {
285  return Vector3D::Colinearity::UNDEFINED;
286  }
287  Vector3D combined = (*this) + vector_b;
288  double combined_length = combined.length();
289  double added_length = (*this).length() + vector_b.length();
290  if (std::fabs(added_length - combined_length) < EPSILON) {
291  return Vector3D::Colinearity::COLINEAR_PRO;
292  }
293  double subtracted_length = (*this).length() - vector_b.length();
294  if (std::fabs(subtracted_length - combined_length) < EPSILON) {
295  return Vector3D::Colinearity::COLINEAR_ANTI;
296  }
297  return Vector3D::Colinearity::NO;
298 }

301 Vector3D Vector3D::rotate_about(double angle, const Vector3D& axis) const {
302  // This solution is derived from this page: http://math.stackexchange.com/questions/511370/how-to-rotate-one-vector-about-another
303  double b_dot_b = axis.dot(axis);
304  Vector3D a_component_b_dir = ((*this).dot(axis) * b_dot_b) * axis;
305  Vector3D a_component_b_orth = (*this) - (((*this).dot(axis) / b_dot_b) * axis);
306  Vector3D w = axis.cross(a_component_b_orth);
307  double x1 = cos(angle) / a_component_b_orth.length();
308  double x2 = sin(angle) / w.length();
309  Vector3D a_component_b_orth_rot = a_component_b_orth.length() * ((x1 * a_component_b_orth) + (x2 * w));
310  return a_component_b_orth_rot + a_component_b_dir;
311 }

444 Point3D Point3D::intersection_between_lines(const Point3D& point_a, const Vector3D& vector_a, const Point3D& point_b, const Vector3D& vector_b, bool return_colinear) {
445  if (vector_a.length() < EPSILON || vector_b.length() < EPSILON) {
446  throw std::invalid_argument("CW::Point3D::intersection_between_lines() - cannot find intersection of lines with zero length");
447  }
448  // Find the closest points according to this solution: http://paulbourke.net/geometry/pointlineplane/
449  // And this: http://stackoverflow.com/questions/563198/how-do-you-detect-where-two-line-segments-intersect
450  Vector3D a_to_b = point_b - point_a;
451  Vector3D vec_a_cross_b = vector_a.unit().cross(vector_b.unit());
452  Vector3D a_to_b_cross_vec_a = a_to_b.unit().cross(vector_a.unit());
453  Vector3D zero_vector(0.0,0.0,0.0);
454  // Check for collinearity
455  if (vec_a_cross_b == zero_vector) {
456  if (a_to_b_cross_vec_a == zero_vector) {
457  // Lines are collinear, so there is no intersection
458  if (!return_colinear) {
459  return Point3D(false);
460  }
461  bool opposite_direction;
462  if (vector_a.unit() == vector_b.unit())
463  opposite_direction = false;
464  else
465  opposite_direction = true;
466  double vec_a_factor_0 = a_to_b.dot(vector_a) / vector_a.dot(vector_a);
467  double vec_a_factor_1 = vec_a_factor_0 + (vector_b.dot(vector_a) / vector_a.dot(vector_a));
468  double vec_a_epsilon = Vector3D::EPSILON / vector_a.length(); // Note accuracy needs to be determined
469  if ((!opposite_direction && vec_a_factor_0 < vec_a_epsilon && vec_a_factor_1 > -vec_a_epsilon) ||
470  (opposite_direction && vec_a_factor_1 < vec_a_epsilon && vec_a_factor_0 > -vec_a_epsilon)) {
471  // The intersection is at the start of line A
472  return point_a;
473  }
474  if ((!opposite_direction && vec_a_factor_0 > -vec_a_epsilon && vec_a_factor_0 < (1 + vec_a_epsilon))) {
475  // The intersection is at the start of line B
476  return point_b;
477  }
478  if (opposite_direction && vec_a_factor_1 > -vec_a_epsilon && vec_a_factor_1 < (1 + vec_a_epsilon)) {
479  // The intersection is at the end of line B
480  return point_b + vector_b;
481  }
482  // Lines are disjoint
483  return Point3D(false);
484 
485  }
486  else {
487  // Lines are parallel, and so there is no intersection.
488  return Point3D(false);
489  }
490  }
491  // We create lines and find the closest points between them.
492  Line3D line_a(point_a, vector_a);
493  Line3D line_b(point_b, vector_b);
494  std::pair<Point3D, Point3D> closest_points = line_a.closest_points(line_b);
495  // Check closest points are actually the same.
496  if (closest_points.first != closest_points.second) {
497  // No intersection
498  return Point3D(false);
499  }
500  // Otherwise, we need to see if the intersection of the lines lie between the line segments
501  Point3D intersection = closest_points.first;
502  Vector3D a_to_int = intersection - point_a;
503  Vector3D b_to_int = intersection - point_b;
504  // Express the point of intersection as a scalar of the vectors from point A and point B
505  double a_direction_factor = 0.0;
506  if (a_to_int != zero_vector)
507  a_direction_factor = vector_a.unit().dot(a_to_int.unit());
508  double b_direction_factor = 0.0;
509  if (b_to_int != zero_vector)
510  b_direction_factor = vector_b.unit().dot(b_to_int.unit());
511 
512  double a_factor = a_direction_factor * a_to_int.length() / vector_a.length();
513  double b_factor = b_direction_factor * b_to_int.length() / vector_b.length();
514  double a_epsilon = Vector3D::EPSILON / vector_a.length();
515  double b_epsilon = Vector3D::EPSILON / vector_b.length();
516  if (a_factor > -a_epsilon && a_factor < 1.0 + a_epsilon &&
517  b_factor > -b_epsilon && b_factor < 1.0 + b_epsilon) {
518  return intersection;
519  }
520  return Point3D(false);
521 }

524 Point3D Point3D::ray_line_intersection(const Point3D& point_a, const Vector3D& vector_a, const Point3D& point_b, const Vector3D& ray_b, bool return_colinear) {
525  // Solution to finding intersection between two line segments derived from this answer: http://stackoverflow.com/questions/563198/how-do-you-detect-where-two-line-segments-intersect
526  Vector3D zero_vector(0.0,0.0,0.0);
527  if (ray_b == zero_vector || vector_a == zero_vector) {
528  throw std::invalid_argument("CW::Point3D::ray_line_intersection() - given ray/line have zero length");
529  }
530  Vector3D a_to_b = point_b - point_a;
531  // Check if the start of the lines touch
532  if (a_to_b == zero_vector) {
533  return point_a;
534  }
535  Vector3D b_to_a = -a_to_b;
536  Vector3D vec_a_cross_b_z_comp = vector_a.unit().cross(ray_b.unit()); // Use unit vectors for easier comparisons with zero vector
537  Vector3D a_to_b_cross_vec_a_z_comp = a_to_b.unit().cross(vector_a.unit()); // Use unit vectors for easier comparisons with zero vector
538  // Check for collinearity
539  if (vec_a_cross_b_z_comp == zero_vector) {
540  if (a_to_b_cross_vec_a_z_comp == zero_vector) {
541  // Lines are collinear, so there is no intersection
542  if (!return_colinear) {
543  return Point3D(false);
544  }
545  double b_to_a_factor = b_to_a.dot(ray_b) / ray_b.dot(ray_b);
546  double b_to_a_end_factor = b_to_a_factor + (vector_a.dot(ray_b) / ray_b.dot(ray_b));
547  if ((b_to_a_factor < 0.0 && b_to_a_end_factor > 0.0) || (b_to_a_factor > 0.0 && b_to_a_end_factor < 0.0)) {
548  // The start of ray B is on line A
549  return point_b;
550  }
551  if (b_to_a_factor > 0.0) {
552  if (b_to_a_factor < b_to_a_end_factor) {
553  // The intersection is at the start of line A
554  return point_a;
555  }
556  else {
557  // The intersection is at the end of line B
558  return point_a + vector_a;
559  }
560  }
561  // Lines are disjoint
562  return Point3D(false);
563 
564  }
565  else {
566  // Lines are parallel, and so there is no intersection.
567  return Point3D(false);
568  }
569  }
570  // We create lines and find the closest points between them.
571  Line3D line_a(point_a, vector_a);
572  Line3D line_b(point_b, ray_b);
573  std::pair<Point3D, Point3D> closest_points = line_a.closest_points(line_b);
574  // Check closest points are actually the same.
575  if (closest_points.first != closest_points.second) {
576  // No intersection
577  return Point3D(false);
578  }
579  // Otherwise, we need to see if the intersection of the lines lie between the line segments
580  Point3D intersection = closest_points.first;
581  Vector3D a_to_int = intersection - point_a;
582  Vector3D b_to_int = intersection - point_b;
583  // Express the point of intersection as a scalar of the vectors from point A and point B
584  double a_direction_factor = 0.0;
585  if (a_to_int != zero_vector)
586  a_direction_factor = vector_a.unit().dot(a_to_int.unit());
587  double b_direction_factor = 0.0;
588  if (b_to_int != zero_vector)
589  b_direction_factor = ray_b.unit().dot(b_to_int.unit());
590 
591  double a_factor = a_direction_factor * a_to_int.length() / vector_a.length();
592  double b_factor = b_direction_factor * b_to_int.length() / ray_b.length();
593  double a_epsilon = Vector3D::EPSILON / vector_a.length();
594  double b_epsilon = Vector3D::EPSILON / ray_b.length();
595  if (a_factor > -a_epsilon && a_factor < 1.0 + a_epsilon &&
596  b_factor > -b_epsilon) {
597  return intersection;
598  }
599  return Point3D(false);

1047 std::pair<Point3D, Point3D> Line3D::closest_points(const Line3D &other_line) const {
1048  if (!other_line) {
1049  throw std::invalid_argument("CW::Line3D::closest_points(): given line is null");
1050  }
1051  if (!(*this)) {
1052  throw std::logic_error("CW::Line3D::closest_points(): this line is null");
1053  }
1054 
1055  // @see http://paulbourke.net/geometry/pointlineplane/
1056  double d1343,d4321,d1321,d4343,d2121;
1057  double numer,denom;
1058  // p13 is a vector between two points on different lines
1059  Vector3D p13 = this->point - other_line.point;
1060  Vector3D p43 = other_line.direction;
1061  if (std::abs(p43.x) < EPSILON && std::abs(p43.y) < EPSILON && std::abs(p43.z) < EPSILON) {
1062  // The two lines are along each other
1063  return std::pair<Point3D,Point3D> (Point3D(false), Point3D(false));
1064  }
1065  Vector3D p21 = this->direction;
1066  //if (std::abs(p21.x) < EPSILON/2 && std::abs(p21.y) < EPSILON/2 && std::abs(p21.z) < EPSILON/2) {
1067  // The vector is zero - should not happen
1068  // return std::pair<Point3D,Point3D> (Point3D(false), Point3D(false));
1069  //}
1070  d1343 = p13.x * p43.x + p13.y * p43.y + p13.z * p43.z;
1071  d4321 = p43.x * p21.x + p43.y * p21.y + p43.z * p21.z;
1072  d1321 = p13.x * p21.x + p13.y * p21.y + p13.z * p21.z;
1073  d4343 = p43.x * p43.x + p43.y * p43.y + p43.z * p43.z;
1074  d2121 = p21.x * p21.x + p21.y * p21.y + p21.z * p21.z;
1075

1080  }
1081  numer = d1343 * d4321 - d1321 * d4343;
1082

1112 bool Line3D::on_line(const Point3D& test_point) const {
1113  if (!test_point) {
1114  throw std::invalid_argument("CW::Line3D::on_line(): given point is null");
1115  }
1116  if (!(*this)) {
1117  throw std::logic_error("CW::Line3D::on_line(): this line is null");
1118  }
1119  double factor;
1120  enum class OnLineFactorCoords {
1121  X,
1122  Y,
1123  Z
1124  };
1125  OnLineFactorCoords factor_val; // This is to save extra calculations.
1126  // To get the most accurate factor, use one of the larger values among x, y OR z to calculate the factor. The smallest possible value for all three x, y, z to have (given that direction is a unit vector) is 1 / sqrt(3) = 0.577
1127  if (this->direction.x > 0.56) {
1128  // Get the factor with which to multiply the x value, and see if it is the same for y and z values too.
1129  factor = (test_point.x - this->point.x) / this->direction.x;
1130  factor_val = OnLineFactorCoords::X;
1131  }
1132  else if (this->direction.y > 0.56) {
1133  factor = (test_point.y - this->point.y) / this->direction.y;
1134  factor_val = OnLineFactorCoords::Y;
1135  }
1136  else {
1137  factor = (test_point.z - this->point.z) / this->direction.z;
1138  factor_val = OnLineFactorCoords::Z;
1139  }
1140  if (factor_val != OnLineFactorCoords::X) {
1141  double x_test = this->point.x + (this->direction.x * factor);
1142  if (std::abs(x_test - test_point.x) > EPSILON) {
1143  return false;
1144  }
1145  }
1146  if (factor_val != OnLineFactorCoords::Y) {
1147  double y_test = this->point.y + (this->direction.y * factor);
1148  if (std::abs(y_test - test_point.y) > EPSILON) {
1149  return false;
1150  }
1151  }
1152  if (factor_val != OnLineFactorCoords::Z) {
1153  double z_test = this->point.z + (this->direction.z * factor);
1154  if (std::abs(z_test - test_point.z) > EPSILON) {
1155  return false;
1156  }
1157  }
1158  return true;
1159 }
