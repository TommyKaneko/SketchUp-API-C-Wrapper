113 Texture Texture::copy() const {
114  // First create the file.
115  std::string file_path = "/tmp/tmptexture";
116  SUResult res = this->save(file_path);
117  assert(res == SU_ERROR_NONE);
118  Texture new_texture(file_path, s_scale(), t_scale());
119  // Delete the temporary file
120  const char *file_path_c = file_path.c_str();
121  int ret = std::remove(file_path_c);
122  assert(ret == 0);
123  // Reset the file name
124  new_texture.file_name(this->file_name());
125  return new_texture;
126  // @developers_notes Method below is efficient and simple, but fails to transfer the scale of the texture, due to the lack of SUTextureSetDimensions() function in the C API. A workaround method is used for now shown above.

128  ImageRep image_rep = this->image_rep().copy();
129  Texture new_texture(image_rep);
130  new_texture.file_name(this->file_name());
131  return new_texture;
132  */
133 }
