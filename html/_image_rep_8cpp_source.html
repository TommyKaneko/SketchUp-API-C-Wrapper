39 SUImageRepRef ImageRep::copy_reference(const ImageRep& other) {
40  if (other.m_attached || SUIsInvalid(other.m_image_rep)) {
41  return other.m_image_rep;
42  }
43  ImageRep new_copy = other.copy();
44  new_copy.m_attached = true; // Ensures that the object will not be destroyed at the end this scope.
45  return new_copy.ref();
46 }

120 void ImageRep::set_data(size_t width, size_t height, size_t bits_per_pixel, size_t row_padding, const std::vector<SUByte> pixel_data) {
121  if(!(*this)) {
122  throw std::logic_error("CW::ImageRep::set_data(): ImageRep is null");
123  }
124  SUResult res = SUImageRepSetData(m_image_rep, width, height, bits_per_pixel, row_padding, &pixel_data[0]);
125  if (res == SU_ERROR_OUT_OF_RANGE) {
126  if (width == 0 || height == 0) {
127  throw std::invalid_argument("CW::ImageRep::set_data(): given width or height is 0 - it must be greater than 0");
128  }
129  else {
130  throw std::invalid_argument("CW::ImageRep::set_data(): given bits_per_pixel must be 8, 24 or 32");
131  }
132  }
133  assert(res == SU_ERROR_NONE);
134 }
