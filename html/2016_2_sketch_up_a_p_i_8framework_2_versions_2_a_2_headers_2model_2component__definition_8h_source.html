22 /**
23 @struct SUComponentBehavior
24 @brief Describes how the component behaves in the SketchUp rendering scene.
25 */
26 struct SUComponentBehavior {
27  /**
28  @enum SUSnapToBehavior
29  @brief Describes how the component instance is placed when it is first
30  instantiated in the rendering scene. For example a window component
31  instance should snap to a vertical plane when instantiated in the
32  rendering scene.
33  */
34  enum SUSnapToBehavior {
35  SUSnapToBehavior_None = 0,
36  SUSnapToBehavior_Any,
37  SUSnapToBehavior_Horizontal,
38  SUSnapToBehavior_Vertical,
39  SUSnapToBehavior_Sloped
40  };
41  enum SUSnapToBehavior component_snap;
42  bool component_cuts_opening; ///< Whether the component creates an opening
43  ///< when placed on a surface, e.g. a window
44  ///< frame component.
45  bool component_always_face_camera; ///< Whether the component behaves like a
46  ///< billboard, where the component always
47  ///< presents a 2D surface perpendicular
48  ///< to the direction of camera.
49  bool component_shadows_face_sun; ///< Whether the component always casts a
50  ///< shadow as if it were facing the
51  ///< direction of the sun.
52  size_t component_no_scale_mask; ///< Bitmask where set bits indicate which
53  ///< scale tool handles are hidden on a given
54  ///< component:
55  ///< 0: X,
56  ///< 1: Y
57  ///< 2: Z
58  ///< 3: X + Z plane
59  ///< 4: Y + Z plane
60  ///< 5: X + Y plane
61  ///< 6: disable scale uniform (XYZ)
62 };
