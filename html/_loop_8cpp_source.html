56 LoopInput Loop::loop_input() const {
57  if(!(*this)) {
58  throw std::logic_error("CW::Loop::loop_input(): Loop is null");
59  }
60  std::vector<Edge> edges = this->edges();
61  std::vector<InputEdgeProperties> edge_properties;
62  edge_properties.reserve(edges.size());
63  for (size_t i=0; i < edges.size(); i++) {
64  InputEdgeProperties edge_prop;
65  edge_prop.hidden = edges[i].hidden();

67  edge_prop.smooth = edges[i].smooth();
68  edge_prop.material = edges[i].material();
69  edge_prop.layer = edges[i].layer();
70  edge_properties.push_back(edge_prop);
71  }
72  return LoopInput(edge_properties);

88  for (size_t i=0; i < count; ++i) {

157 PointLoopClassify Loop::classify_point(const std::vector<Point3D>& loop_points, const Point3D& test_point) {
158  if(loop_points.size() < 3) {
159  throw std::invalid_argument("CW::Loop::classify_point(): Fewer than 3 points given - not a valid loop.");
160  }
161  if(!test_point) {
162  throw std::invalid_argument("CW::Loop::classify_point(): Point3D given is null");
163  }
164  // http://www.geeksforgeeks.org/how-to-check-if-a-given-point-lies-inside-a-polygon/
165  // First check that the test point is on the plane
166  Plane3D loop_plane = Plane3D::plane_from_loop(loop_points);
167  if (!loop_plane) {
168  throw std::invalid_argument("CW::Loop::classify_point(): Points given does not form a valid loop.");
169  }
170  else if (!loop_plane.on_plane(test_point)) {
171  return PointLoopClassify::PointNotOnPlane;
172  }
173  // Now check if it is on the vertices
174  for (size_t i=0; i < loop_points.size(); i++) {
175  if (loop_points[i] == test_point) {
176  return PointLoopClassify::PointOnVertex;
177  }
178  }
179  // Now check if it is on the edges
180  for (size_t i=0; i < loop_points.size(); i++) {
181  Point3D next_point;
182  if (i == loop_points.size()-1) {
183  next_point = loop_points[0];
184  }
185  else {
186  next_point = loop_points[i+1];
187  }
188  if (Line3D::on_line_segment(loop_points[i], next_point, test_point)) {
189  return PointLoopClassify::PointOnEdge;
190  }
191  }
192  // Now check if it is inside or outside, given that we know that the point is (a) on the same plane of the loop, and (b) not on the edge, and (c) not on the vertices.
193  // From: http://www.geeksforgeeks.org/how-to-check-if-a-given-point-lies-inside-a-polygon/
194  // We draw a line from the point, and if it intersects with the loop an even number of times, then it is outside, if it intersects an odd number of times, it is inside.
195  // The line to draw can be any vector that is coplanar to the loop's plane - the simple one is get a vector of two points within the loop.
196  assert(loop_points[0] != loop_points[1]);
197  Vector3D ray = Vector3D(loop_points[1] - loop_points[0]).unit();
198  // Colinear line segments create problems for the algorith below, so we ignore them by creating a new list of points to calculate that exclude colinear edges.
199  struct LoopPoint {
200  Point3D point;
201  Vector3D vector_to_next;
202  };
203  std::vector<LoopPoint> red_loop_points;
204  red_loop_points.reserve(loop_points.size());
205  for (size_t i=0; i < loop_points.size(); i++) {
206  Point3D next_point;
207  if (i == loop_points.size()-1) {
208  next_point = loop_points[0];
209  } else {
210  next_point = loop_points[i+1];
211  }
212  Vector3D next_vector = next_point - loop_points[i];
213  if (next_vector.unit() != ray.unit() && next_vector.unit() != -ray.unit()) {
214  red_loop_points.push_back(LoopPoint{loop_points[i], next_vector});
215  }
216  }
217  // Now we have a list of points with colinear points removed, find intersections with the line segments
218  size_t num_intersections = 0;
219  for (size_t i=0; i < red_loop_points.size(); i++) {
220  Point3D intersection = Point3D::ray_line_intersection(red_loop_points[i].point, red_loop_points[i].vector_to_next, test_point, ray, false);
221  if (!!intersection) {
222  // There is a case where the ray intersection lies on a vertex of the loop. Whether this should be counted as an intersection of both lines connected to the loop or just once through depends on the orientation of the ray relative to the lines connected to that vertex
223  if (intersection == red_loop_points[i].point) {
224  // Get the directions of the lines connected to the vertex
225  LoopPoint prev_point;
226  if (i == 0) {
227  prev_point = red_loop_points[red_loop_points.size()-1];
228  }
229  else {
230  prev_point = red_loop_points[i-1];
231  }
232  Vector3D prev_cross = ray.cross(prev_point.vector_to_next);
233  Vector3D next_cross = ray.cross(red_loop_points[i].vector_to_next);
234  assert(fabs(next_cross.length()) > Vector3D::EPSILON);
235  if (prev_cross.unit() != next_cross.unit()) {
236  // The cross product direction is in opposite directions - ie the lines connected to the vertex are on the same side of the ray, therefore, count this as an additional intersection
237  num_intersections++;
238  }
239  }
240  else {
241  num_intersections++;
242  }
243  }
244  }
245  if (num_intersections % 2 == 0) {
246  // Even number of intersections
247  return PointLoopClassify::PointOutside;
248  }
249  else {
250  return PointLoopClassify::PointInside;
251  }
252 }
