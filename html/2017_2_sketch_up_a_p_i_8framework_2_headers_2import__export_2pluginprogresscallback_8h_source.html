7 /**
8 @struct SketchUpPluginProgressCallback
9 @brief Interface to provide import/export progress feedback and cancellation.
10 
11 All well behaved SketchUp importer/exporter plugins should support progress
12 feedback and cancellation through this interface. SketchUp will pop up a
13 progress/cancel dialog prior to calling the appropriate Convert method on the
14 importer/exporter plugin and supply a concrete implementation of this interface.
15 The plugin should supply feedback to the user by estimating how far along the
16 conversion is (via SetPercentDone or SetStepSize and Step), provide some textual
17 progress (via SetProgressMessage) and test for cancellation by calling
18 HasBeenCancelled.
19 
20 For example, the main loop of an importer plugin may look like this:
21 
22 \code{.cpp}
23 progress_callback->SetProgressMessage("Importing 50 things!");
24 progress_callback->SetStepSize(1.0);
25 for (int steps = 0; steps < 50; ++steps) {
26  progress_callback->Step();
27  // Do something which builds an skp model
28  if (progress_callback->HasBeenCancelled()) {
29  // Clean up and return
30  return false;
31  }
32 }
33 progress_callback->SetProgressMessage("We're half way there!");
34 // Do some more larger steps
35 progress_callback->SetPercentDone(75.0);
36 progress_callback->SetProgressMessage("Finishing up!");
37 // And some more
38 progress_callback->SetPercentDone(100.0);
39 // Success!
40 return true;
41 \endcode
42 
43 */

46  /**
47  @brief Call this to test if the user has pressed the cancel button. Your
