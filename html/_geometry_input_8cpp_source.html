161 size_t GeometryInput::add_face(const Face &face) {
162  if(!(*this)) {
163  throw std::logic_error("CW::GeometryInput::add_face(): GeometryInput is null");
164  }
165  if(!face) {
166  throw std::invalid_argument("CW::GeometryInput::add_face(): Face argument is null");
167  }
168  SULoopInputRef loop_input = SU_INVALID;
169  SUResult res = SULoopInputCreate(&loop_input);
170  assert(res == SU_ERROR_NONE);
171  // Add outer loop
172  std::vector<Vertex> outer_vertices = face.outer_loop().vertices();
173  std::vector<Edge> outer_edges = face.outer_loop().edges();
174  for (size_t i=0; i < outer_vertices.size(); ++i) {
175  SUPoint3D point = outer_vertices[i].position();
176  res = SUGeometryInputAddVertex(m_geometry_input, &point);
177  assert(res == SU_ERROR_NONE);
178  res = SULoopInputAddVertexIndex(loop_input, m_vertex_index);
179  assert(res == SU_ERROR_NONE);
180  if (outer_edges[i].hidden()) {
181  res = SULoopInputEdgeSetHidden(loop_input, i, true);
182  assert(res == SU_ERROR_NONE);
183  }
184  if (outer_edges[i].smooth()) {
185  res = SULoopInputEdgeSetSmooth(loop_input, i, true);
186  assert(res == SU_ERROR_NONE);
187  }
188  if (outer_edges[i].soft()) {
189  res = SULoopInputEdgeSetSoft(loop_input, i, true);
190  assert(res == SU_ERROR_NONE);
191  }
192  ++m_vertex_index;
193  }
194  size_t added_face_index;
195  res = SUGeometryInputAddFace(m_geometry_input, &loop_input, &added_face_index);
196  assert(res == SU_ERROR_NONE);
197  // Add inner loops
198  std::vector<Loop> inner_loops = face.inner_loops();
199  for (size_t i=0; i < inner_loops.size(); ++i) {
200  SULoopInputRef inner_loop_input = SU_INVALID;
201  res = SULoopInputCreate(&loop_input);
202  assert(res == SU_ERROR_NONE);
203  std::vector<Vertex> inner_vertices = inner_loops[i].vertices();
204  std::vector<Edge> inner_edges = inner_loops[i].edges();
205  for (size_t j=0; j < inner_vertices.size(); ++j) {
206  SUPoint3D point = inner_vertices[j].position();
207  res = SUGeometryInputAddVertex(m_geometry_input, &point);
208  assert(res == SU_ERROR_NONE);
209  res = SULoopInputAddVertexIndex(inner_loop_input, m_vertex_index);
210  assert(res == SU_ERROR_NONE);
211  ++m_vertex_index;
212  if (inner_edges[j].hidden()) {
213  res = SULoopInputEdgeSetHidden(loop_input, j, true);
214  assert(res == SU_ERROR_NONE);
215  }
216  if (inner_edges[j].smooth()) {
217  res = SULoopInputEdgeSetSmooth(loop_input, j, true);
218  assert(res == SU_ERROR_NONE);
219  }
220  if (inner_edges[j].soft()) {
221  res = SULoopInputEdgeSetSoft(loop_input, j, true);
222  assert(res == SU_ERROR_NONE);
223  }
224  }
225  res = SUGeometryInputFaceAddInnerLoop(m_geometry_input, added_face_index, &inner_loop_input);
226  assert(res == SU_ERROR_NONE);
227  }
228  // TODO: deallocate LoopInputRef? It doesn't FEEL necessary because it has been added to the Geometry Input object. Needs testing.
229  // Add layer
230  // TODO: there are problems with releasing a model that gets a layer from another model. The following code causes a EXEC_BAD_ACCESS error when calling SUModelRelease, during CLayerDestroy process.
231  /*
232  Layer face_layer = face.layer();
233  if (!!face_layer) {
234  res = SUGeometryInputFaceSetLayer(m_geometry_input, added_face_index, face_layer.ref());
235  assert(res == SU_ERROR_NONE);
236  }
237  // Set Materials TODO: not done quite right here, I don't think... uv coords are not set.
238  SUMaterialInput material = MaterialInput(face.material()).ref();
239  res = SUGeometryInputFaceSetFrontMaterial(m_geometry_input, added_face_index, &material);
240  assert(res == SU_ERROR_NONE);
241  SUMaterialInput back_material = MaterialInput(face.back_material()).ref();
242  res = SUGeometryInputFaceSetBackMaterial(m_geometry_input, added_face_index, &back_material);
243  assert(res == SU_ERROR_NONE);
244  */
245  // There are further properties that need to be added, such as attributes, and these must be added after the entities::fill() operation.
246  m_faces.push_back(std::pair<size_t, Face> (added_face_index, face));
247  ++m_num_faces;
248  return added_face_index;
249 }

264 size_t GeometryInput::add_edge(const Edge &edge) {
265  if(!edge) {
266  throw std::invalid_argument("CW::GeometryInput::add_edge(): Edge argument is null");
267  }
268  if (SU_API_VERSION_MAJOR < 5) {
269  // Edges can only be added since SU2017
270  return 0;
271  }
272  size_t start_vertex_index = m_vertex_index;
273  SUPoint3D start_point = edge.start().position();
274  SUResult res = SUGeometryInputAddVertex(m_geometry_input, &start_point);
275  m_vertex_index++;
276  assert(res == SU_ERROR_NONE);
277  size_t end_vertex_index = m_vertex_index;
278  SUPoint3D end_point = edge.end().position();
279  res = SUGeometryInputAddVertex(m_geometry_input, &end_point);
280  m_vertex_index++;
281  assert(res == SU_ERROR_NONE);
282  size_t added_edge_index;
283  res = SUGeometryInputAddEdge(m_geometry_input, start_vertex_index, end_vertex_index, &added_edge_index);
284  assert(res == SU_ERROR_NONE);
285 
286  // Add other information about the edge
287  if (!!edge.material()) {
288  res = SUGeometryInputEdgeSetMaterial(m_geometry_input, added_edge_index, edge.material().ref());
289  assert(res == SU_ERROR_NONE);
290  }
291  if (edge.hidden()) {
292  res = SUGeometryInputEdgeSetHidden(m_geometry_input, added_edge_index, true);
293  assert(res == SU_ERROR_NONE);
294  }
295  if (edge.smooth()) {
296  res = SUGeometryInputEdgeSetSmooth(m_geometry_input, added_edge_index, true);
297  assert(res == SU_ERROR_NONE);
298  }
299  if (edge.soft()) {
300  res = SUGeometryInputEdgeSetSoft(m_geometry_input, added_edge_index, true);
301  assert(res == SU_ERROR_NONE);
302  }
303  m_edges.push_back(std::pair<size_t, Edge> (added_edge_index, edge));
304  ++m_num_edges;
305  return added_edge_index;
306 }
