56 SUFaceRef Face::create_face(std::vector<Point3D>& outer_points, LoopInput& loop_input) {
57  SUFaceRef face = SU_INVALID;
58  SULoopInputRef loop_input_ref = loop_input.ref();
59  std::vector<SUPoint3D> su_points(outer_points.size());
60  std::transform(outer_points.begin(), outer_points.end(), su_points.begin(),
61  [](const Point3D& value){
62  return (SUPoint3D)value;
63  });
64  SUResult res = SUFaceCreate(&face, su_points.data(), &loop_input_ref);
65  if (res != SU_ERROR_NONE) {
66  // The points cannot be made into a face: either the points do not lie in a plane, or is somehow problematic.
67  return SU_INVALID;
68  }
69  loop_input.m_attached = true;
70  return face;
71 }

74 SUFaceRef Face::copy_reference(const Face& other) {
75  if (other.m_attached || !other) {
76  return other.ref();
77  }
78  // The other face has not been attached to the model, so copy its properties to a new object
79  Loop other_outer_loop = other.outer_loop();
80  std::vector<Point3D> other_outer_points = other_outer_loop.points();
81  LoopInput outer_loop_input = other_outer_loop.loop_input();
82  SUFaceRef new_face = create_face(other_outer_points, outer_loop_input);
83  return new_face;
84 }

220 /*
221 * determine if a given Point3d is on the referenced Face. The return value is calculated from this list:
222  PointUnknown (indicates an error),
223  PointInside (point is on the face, not in a hole),
224  PointOnVertex (point touches a vertex),
225  PointOnEdge (point is on an edge),
226  PointOutside (point outside the face or in a hole),
227  PointNotOnPlane (point off the face's plane).
228 * @param SUPoint3D object.
229 * @return FacePointClass enum indicating the status of the point relative to the face.
230 */
231 FacePointClass Face::classify_point(const Point3D& point) {
232  if (!(*this)) {
233  throw std::logic_error("CW::Face::classify_point(): Face is null");
234  }
235  if (!point) {
236  throw std::invalid_argument("CW::Face::classify_point(): Given Point3D object is null");
237  }
238  Loop outer_loop = this->outer_loop();
239  PointLoopClassify outer_class = outer_loop.classify_point(point);
240  switch (outer_class) {
241  case (PointLoopClassify::PointUnknown):
242  return FacePointClass::PointUnknown;
243  case (PointLoopClassify::PointNotOnPlane):
244  return FacePointClass::PointNotOnPlane;
245  case (PointLoopClassify::PointOnEdge):
246  return FacePointClass::PointOnEdge;
247  case (PointLoopClassify::PointOnVertex):
248  return FacePointClass::PointOnVertex;
249  case (PointLoopClassify::PointOutside):
250  return FacePointClass::PointOutside;
251  case (PointLoopClassify::PointInside): {
252  std::vector<Loop> inner_loops = this->inner_loops();
253  for (size_t i=0; i < inner_loops.size(); i++) {
254  PointLoopClassify inner_class = inner_loops[i].classify_point(point);
255  switch (inner_class) {
256  case (PointLoopClassify::PointOnEdge):
257  return FacePointClass::PointOnEdge;
258  case (PointLoopClassify::PointOnVertex):
259  return FacePointClass::PointOnVertex;
260  case (PointLoopClassify::PointOutside):
261  break;
262  case (PointLoopClassify::PointInside):
263  return FacePointClass::PointOutside;
264  break;
265  case (PointLoopClassify::PointNotOnPlane):
266  case (PointLoopClassify::PointUnknown):
267  assert(false); // this should not happen
268  }
269  }
270  return FacePointClass::PointOutside;
271  } break;
272  }
273  return FacePointClass::PointUnknown;
274 }

277 std::vector<Edge> Face::edges() {
278  if (!(*this)) {
279  throw std::logic_error("CW::Face::edges(): Face is null");
280  }
281  std::vector<Edge> total_edges = outer_loop().edges();
282  std::vector<Loop> all_loops = loops();
283  size_t total_edge_num = 0;
284  for (size_t i=0; i < all_loops.size(); i++) {
285  total_edge_num += all_loops[i].size();
286  }
287  total_edges.reserve(total_edge_num);
288  for (size_t i=0; i < all_loops.size(); i++) {
289  std::vector<Edge> loop_edges = all_loops[i].edges();
290  total_edges.insert(total_edges.end(), loop_edges.begin(), loop_edges.end());
291  }
292  return total_edges;
293 }
