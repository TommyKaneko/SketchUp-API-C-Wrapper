43 SUComponentInstanceRef ComponentInstance::copy_reference(const ComponentInstance& other) {
44  if (other.m_attached) {
45  return other.ref();
46  }
47  // The other face has not been attached to the model, so copy its properties to a new object
48  ComponentInstance new_instance = other.definition().create_instance();
49  new_instance.transformation(other.transformation());
50  new_instance.name(other.name());
51  return new_instance.ref();
52 }
