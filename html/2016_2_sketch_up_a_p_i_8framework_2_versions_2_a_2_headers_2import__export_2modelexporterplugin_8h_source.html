16 class SketchUpModelExporterInterface {
17  public:
18  /**
19  @brief Returns a unique ASCII (non-localized) identifier for the exporter.
20  For example, all SketchUp supplied exporters use the following
21  convention for the id name: com.sketchup.exporters.extension. This id
22  is used to identify the current default exporter and saved between
23  invocations of SketchUp.
24  @return
25  - A unique string identifier for the plugin
26  */
27  virtual std::string GetIdentifier() const = 0;

29  /**
30  @brief The number of extensions supported by the exporter. Must be > 0.
31  @return
32  - The number of file extensions supported by this plugin
33  */
34  virtual int GetFileExtensionCount() const = 0;
35 
36  /**
37  @brief Each extension, assumed to be ASCII and case independent with no
38  leading dot.
39  @param[in] index The index of the extension
40  @return
41  - The file extension retrieved
42  */
43  virtual std::string GetFileExtension(int index) const = 0;
44 
45  /**
46  @brief Return a brief description of the exporter for the file drop down box
47  which may be extension specific. For example you may return "AutoCAD
48  DWG Files" corresponding to the dwg extension and "AutoCAD DXF Files"
49  corresponding to the dxf extension. Each extension and description pair
50  will be entered into the exporter drop down list as a separate entry.
51  @param[in] index The index of the extension
52  @return
53  - The description retrieved
54  */
55  virtual std::string GetDescription(int index) const = 0;

58  /**
59  @brief Indicates whether the exporter supports an options dialog.
60  @return
61  - true if the exporter supports an options dialog. If so, the "Options" button
62  will be enabled when your exporter is chosen, and selecting that button will
63  call the exporter's ShowOptionsDialog method.
64  */
65  virtual bool SupportsOptions() const {
66  return false;
67  }
68 
69  /**
70  @brief Displays a modal dialog which controls options supported by the
71  exporter. Implementing this is required if SupportOptions is true.
72  Options should be saved someplace which persists between this method
73  and ConvertFromSkp as well as between sessions.
74  @param[in] model_has_selection Indicates if the model has a current selection
75  set. May be used in concert with ExportSelectionSetOnly to allow
76  the exporter to export just the selected portion of the model.
77  For example, you may enable a check box which controls this and
78  return the state of that check box from ExportSelectionSetOnly.
79  */
80  virtual void ShowOptionsDialog(bool model_has_selection) {}
81 
82  /**
83  @brief Indicates whether the plugin supports exporting just the selection.
84  @return
85  - true if the exporter supports an option to just export the selection AND the
86  user has set that to true. If true, SketchUp will save the selected portion
87  of the model and pass that into the ConvertFromSkp method.
88  */
89  virtual bool ExportSelectionSetOnly() {
90  return false;
91  }
92 
93  /**
94  @brief Indicates whether the plugin supports the progress callback.
95  @return
96  - true if the exporter supports the progress callback. If true, the plugin
97  should periodically update the status with messages and percent complete
98  as well as check if the user pressed the cancel button. If false, SketchUp
99  will provide limited feedback on the progress of the export step.
100  */
101  virtual bool SupportsProgress() const { return true; }
102 
103  /**
104  @brief Performs the conversion from a temporary skp file to the target output
105  file using options set during the ShowOptionsDialog method.
106  @param[in] input_skp Either a saved version of the current model or the
107  current selection (if ExportSelectionSetOnly returns
108  true). The method should read this file and convert to
109  the output file. The file name is specified in UTF-8 and
110  may require conversion to a platform-dependent format
111  (i.e. UTF-16) for proper support of international file
112  names.
113  @param[in] output The requested output file in UTF-8.
114  @param[in] progress The exporter should support this progress interface. See
115  that for details. The method should be tolerant to a NULL
116  progress callback.
117  @param[in] reserved Reserved for internal use.
118  @return
119  - true on success
120  - false on failure or cancellation
121  */
122  virtual bool ConvertFromSkp(const std::string& input_skp,
123  const std::string& output,
124  SketchUpPluginProgressCallback* progress,
125  void* reserved) = 0;
126  /**
127  @brief Displays a modal dialog showing an optional summary of the export
128  process. The default does nothing. Note that Ruby scripting often
129  disables this dialog.
130  */
131  virtual void ShowSummaryDialog() {}
132 };
