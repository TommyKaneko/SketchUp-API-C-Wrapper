44 SUAxesRef Axes::create_custom_axes(const SUPoint3D& origin, const SUVector3D& xaxis, const SUVector3D& yaxis, const SUVector3D& zaxis) {
45  SUAxesRef axes = SU_INVALID;
46  SUResult res = SUAxesCreateCustom(&axes, &origin, &xaxis, &yaxis, &zaxis);
47  if(res != SU_ERROR_NONE) {
48  // There is a problem with this axis, so return an invalid one
49  if (SUIsValid(axes)) {
50  SUAxesRelease(&axes);
51  }
52  return SU_INVALID;
53  }
54  return axes;
55 }

57 SUAxesRef Axes::copy_reference(const Axes& other) {
58  if (other.m_attached) {
59  return other.m_axes;
60  }
61  // The other axes has not been attached to the model, so copy its properties to a new object
62  SUAxesRef new_axes = create_custom_axes(other.origin(), other.x_axis(), other.y_axis(), other.z_axis());
63  return new_axes;
64 }
