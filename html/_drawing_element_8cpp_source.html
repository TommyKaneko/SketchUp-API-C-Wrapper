87 bool DrawingElement::copy_properties_from(const DrawingElement& element) {
88  bool success = this->casts_shadows(element.casts_shadows());
89  if (!success)
90  return false;
91  success = this->hidden(element.hidden());
92  if (!success)
93  return false;
94  success = this->receive_shadows(element.receive_shadows());
95  if (!success)
96  return false;
97  Layer elem_layer = element.layer();
98  success = this->layer(elem_layer);
99  if (!success)
100  return false;
101  success = this->material(element.material());
102  if (!success)
103  return false;
104 
105  return true;
106 }
