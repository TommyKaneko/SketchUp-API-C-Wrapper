138 /**
139 @brief Exports the document to a .pdf file at the indicated path. If a valid
140  options dictionary object is provided, then it can specify additional
141  PDF export options. The following options (declared in
142  documentexportoptions.h) may be specified in the options dictionary to
143  override their default values:
144 - <b>key: \ref LOExportOption_StartPage, value: Int32</b>\n
145  Start page index. Defaults to 0.
146 - <b>key: \ref LOExportOption_EndPage, value: Int32</b>\n
147  End page index. Defaults to the index of the last page in the document.
148 - <b>key: \ref LOExportOption_OutputResolution, value: Int32</b>\n
149  Output resolution for images (including raster models). Must be one of the
150  values of the \ref LOImageResolution enum. Defaults to the document's current
151  output resolution setting (see \ref LOPageInfoGetOutputResolution).
152 - <b>key: \ref LOExportOption_CompressImages, value: Bool</b>\n
153  If true, images (including raster models) will be compressed using JPEG
154  compression to reduce PDF file size. Defaults to true.
155 - <b>key: \ref LOExportOption_ImageCompressionQuality, value: Double</b>\n
156  Specifies the image compression quality from 0.0 to 1.0, to be used if
157  the \ref LOExportOption_CompressImages option is set to true. Defaults to
158  0.5.
159 
160 @param[in] document The document object.
161 @param[in] path The file path where the pdf should be exported.
162 @param[in] options_dict The options dictionary. If the dictionary object is
163  invalid then default options will be used.
164 @return
165 - \ref SU_ERROR_NONE on success
166 - \ref SU_ERROR_INVALID_INPUT if document does not refer to a valid object
167 - \ref SU_ERROR_NULL_POINTER_INPUT if path is NULL
168 - \ref SU_ERROR_GENERIC if an option was specified using a value of the wrong
169  type
170 - \ref SU_ERROR_OUT_OF_RANGE if the \ref LOExportOption_StartPage option
171  specifies a start page that is out of range
172 - \ref SU_ERROR_OUT_OF_RANGE if the \ref LOExportOption_EndPage option
173  specifies an end page that is out of range
174 - \ref SU_ERROR_OUT_OF_RANGE if the \ref LOExportOption_OutputResolution option
175  specifies a value that is out of range
176 - \ref SU_ERROR_OUT_OF_RANGE if the \ref LOExportOption_ImageCompressionQuality
177  option specifies a value that is out of range
178 - \ref SU_ERROR_SERIALIZATION if there was an error writing the file
179 */
180 LO_RESULT LODocumentExportToPDF(LODocumentRef document,
181  const char* path,
182  LODictionaryRef options_dict);
