86 double Transformation::determinant() const {
87  // From: http://www.euclideanspace.com/maths/algebra/matrix/functions/determinant/fourD/index.htm
88  // Note this is for general 4x4 matrix, and it is excessive for a 3x3 matrix.
89  double value =
90  m_transformation.values[12] * m_transformation.values[9] * m_transformation.values[6] * m_transformation.values[3]-m_transformation.values[8] * m_transformation.values[13] * m_transformation.values[6] * m_transformation.values[3]-m_transformation.values[12] * m_transformation.values[5] * m_transformation.values[10] * m_transformation.values[3]+m_transformation.values[4] * m_transformation.values[13] * m_transformation.values[10] * m_transformation.values[3]+
91  m_transformation.values[8] * m_transformation.values[5] * m_transformation.values[14] * m_transformation.values[3]-m_transformation.values[4] * m_transformation.values[9] * m_transformation.values[14] * m_transformation.values[3]-m_transformation.values[12] * m_transformation.values[9] * m_transformation.values[2] * m_transformation.values[7]+m_transformation.values[8] * m_transformation.values[13] * m_transformation.values[2] * m_transformation.values[7]+
92  m_transformation.values[12] * m_transformation.values[1] * m_transformation.values[10] * m_transformation.values[7]-m_transformation.values[0] * m_transformation.values[13] * m_transformation.values[10] * m_transformation.values[7]-m_transformation.values[8] * m_transformation.values[1] * m_transformation.values[14] * m_transformation.values[7]+m_transformation.values[0] * m_transformation.values[9] * m_transformation.values[14] * m_transformation.values[7]+
93  m_transformation.values[12] * m_transformation.values[5] * m_transformation.values[2] * m_transformation.values[11]-m_transformation.values[4] * m_transformation.values[13] * m_transformation.values[2] * m_transformation.values[11]-m_transformation.values[12] * m_transformation.values[1] * m_transformation.values[6] * m_transformation.values[11]+m_transformation.values[0] * m_transformation.values[13] * m_transformation.values[6] * m_transformation.values[11]+
94  m_transformation.values[4] * m_transformation.values[1] * m_transformation.values[14] * m_transformation.values[11]-m_transformation.values[0] * m_transformation.values[5] * m_transformation.values[14] * m_transformation.values[11]-m_transformation.values[8] * m_transformation.values[5] * m_transformation.values[2] * m_transformation.values[15]+m_transformation.values[4] * m_transformation.values[9] * m_transformation.values[2] * m_transformation.values[15]+
95  m_transformation.values[8] * m_transformation.values[1] * m_transformation.values[6] * m_transformation.values[15]-m_transformation.values[0] * m_transformation.values[9] * m_transformation.values[6] * m_transformation.values[15]-m_transformation.values[4] * m_transformation.values[1] * m_transformation.values[10] * m_transformation.values[15]+m_transformation.values[0] * m_transformation.values[5] * m_transformation.values[10] * m_transformation.values[15];
96  return value;
97 }

100 Transformation Transformation::matrix_inverse() const {
101  // Method for transformation is illustrated here: http://stackoverflow.com/questions/2624422/efficient-4x4-matrix-inverse-affine-transform
102  const double& i00 = m_transformation.values[0];
103  const double& i01 = m_transformation.values[1];
104  const double& i02 = m_transformation.values[2];
105  const double& i03 = m_transformation.values[3];
106  const double& i10 = m_transformation.values[4];
107  const double& i11 = m_transformation.values[5];
108  const double& i12 = m_transformation.values[6];
109  const double& i13 = m_transformation.values[7];
110  const double& i20 = m_transformation.values[8];
111  const double& i21 = m_transformation.values[9];
112  const double& i22 = m_transformation.values[10];
113  const double& i23 = m_transformation.values[11];
114  const double& i30 = m_transformation.values[12];
115  const double& i31 = m_transformation.values[13];
116  const double& i32 = m_transformation.values[14];
117  const double& i33 = m_transformation.values[15];
118 
119  double s0 = i00 * i11 - i10 * i01;
120  double s1 = i00 * i12 - i10 * i02;
121  double s2 = i00 * i13 - i10 * i03;
122  double s3 = i01 * i12 - i11 * i02;
123  double s4 = i01 * i13 - i11 * i03;
124  double s5 = i02 * i13 - i12 * i03;
125 
126  double c5 = i22 * i33 - i32 * i23;
127  double c4 = i21 * i33 - i31 * i23;
128  double c3 = i21 * i32 - i31 * i22;
129  double c2 = i20 * i33 - i30 * i23;
130  double c1 = i20 * i32 - i30 * i22;
131  double c0 = i20 * i31 - i30 * i21;
132 
133  double det = s0 * c5 - s1 * c4 + s2 * c3 + s3 * c2 - s4 * c1 + s5 * c0;
134  assert(det != 0.0);
135  double invdet = 1 / det;
136 
137  SUTransformation inverted_t;
138  inverted_t.values[0] = (i11 * c5 - i12 * c4 + i13 * c3) * invdet;
139  inverted_t.values[1] = (-i01 * c5 + i02 * c4 - i03 * c3) * invdet;
140  inverted_t.values[2] = (i31 * s5 - i32 * s4 + i33 * s3) * invdet;
141  inverted_t.values[3] = (-i21 * s5 + i22 * s4 - i23 * s3) * invdet;
142 
143  inverted_t.values[4] = (-i10 * c5 + i12 * c2 - i13 * c1) * invdet;
144  inverted_t.values[5] = (i00 * c5 - i02 * c2 + i03 * c1) * invdet;
145  inverted_t.values[6] = (-i30 * s5 + i32 * s2 - i33 * s1) * invdet;
146  inverted_t.values[7] = (i20 * s5 - i22 * s2 + i23 * s1) * invdet;
147 
148  inverted_t.values[8] = (i10 * c4 - i11 * c2 + i13 * c0) * invdet;
149  inverted_t.values[9] = (-i00 * c4 + i01 * c2 - i03 * c0) * invdet;
150  inverted_t.values[10] = (i30 * s4 - i31 * s2 + i33 * s0) * invdet;
151  inverted_t.values[11] = (-i20 * s4 + i21 * s2 - i23 * s0) * invdet;
152 
153  inverted_t.values[12] = (-i10 * c3 + i11 * c1 - i12 * c0) * invdet;
154  inverted_t.values[13] = (i00 * c3 - i01 * c1 + i02 * c0) * invdet;
155  inverted_t.values[14] = (-i30 * s3 + i31 * s1 - i32 * s0) * invdet;
156  inverted_t.values[15] = (i20 * s3 - i21 * s1 + i22 * s0) * invdet;
157 
158  return Transformation(inverted_t);
159 }

210 Transformation Transformation::inverse() const {
211  if (m_transformation.values[3] != 0.0 ||
212  m_transformation.values[7] != 0.0 ||
213  m_transformation.values[11] != 0.0 ||
214  m_transformation.values[15] != 1.0) {
215  return this->matrix_inverse();
216  }
217 
218  SUTransformation inverted_t;
219  // Method for transformation is illustrated here: http://stackoverflow.com/questions/2624422/efficient-4x4-matrix-inverse-affine-transform
220  /*
221  M = [ R T ]
222  [ 0 W ]
223  where M is 4x4, R is 3x3, T is 3x1, and the bottom row is (0,0,0,1), then
224  inv(M) = [ inv(R) -inv(R) * T ]
225  [ 0 1 ]
226  // First invert the 3x3 sub matrix R (the rotation part)
227  */
228  // [ 0 4 8 ]
229  // [ 1 5 9 ]
230  // [ 2 6 10]
231  // matrix value = ((1. matrix of minors) * 2. cofactor, 3. swap_diagonals) / 4. determinant
232  /*
233  double matrix_minors3x3[9];
234  // 1. Get matrix of minors
235  matrix_minors3x3[0] = ((m_transformation.values[5] * m_transformation.values[10]) - (m_transformation.values[6] * m_transformation.values[9]));
236  matrix_minors3x3[1] = ((m_transformation.values[4] * m_transformation.values[10]) - (m_transformation.values[6] * m_transformation.values[8]));
237  matrix_minors3x3[2] = ((m_transformation.values[4] * m_transformation.values[9]) - (m_transformation.values[5] * m_transformation.values[8]));
238  matrix_minors3x3[3] = ((m_transformation.values[1] * m_transformation.values[10]) - (m_transformation.values[2] * m_transformation.values[9]));
239  matrix_minors3x3[4] = ((m_transformation.values[0] * m_transformation.values[10]) - (m_transformation.values[2] * m_transformation.values[8]));
240  matrix_minors3x3[5] = ((m_transformation.values[0] * m_transformation.values[9]) - (m_transformation.values[1] * m_transformation.values[8]));
241  matrix_minors3x3[6] = ((m_transformation.values[1] * m_transformation.values[6]) - (m_transformation.values[2] * m_transformation.values[5]));
242  matrix_minors3x3[7] = ((m_transformation.values[0] * m_transformation.values[6]) - (m_transformation.values[2] * m_transformation.values[4]));
243  matrix_minors3x3[8] = ((m_transformation.values[1] * m_transformation.values[5]) - (m_transformation.values[1] * m_transformation.values[4]));
244  b[2, 2] = ( values[12]a[3, 0] * s4 - a[3, 1] * s2 + a[3, 3] * s0) * invdet;
245  matrix_minors3x3[8] = ((m_transformation.values[1] * m_transformation.values[5]) - (m_transformation.values[1] * m_transformation.values[4]));
246  // 4. find determinant
247  double determinant = (m_transformation.values[0] * matrix_minors3x3[0]) -
248  (m_transformation.values[4] * matrix_minors3x3[3]) +
249  (m_transformation.values[8] * matrix_minors3x3[6]);
250  assert(determinant != 0.0);
251  // 2. 3., 4. - create matrix of cofactors, adjudicate and divide by determinant.
252  inverted_t.values[0] = matrix_minors3x3[0] / determinant;
253  inverted_t.values[1] = matrix_minors3x3[3] * (-1.0) / determinant; // swapped with 4
254  inverted_t.values[2] = matrix_minors3x3[6] / determinant; //swapped with 8
255  inverted_t.values[4] = matrix_minors3x3[1] * (-1.0) / determinant; // swapped with 1
256  inverted_t.values[5] = matrix_minors3x3[4] / determinant;
257  inverted_t.values[6] = matrix_minors3x3[7] * (-1.0) / determinant; // swapped with 9
258  inverted_t.values[8] = matrix_minors3x3[2] / determinant; // swaped with 2
259  inverted_t.values[9] = matrix_minors3x3[5] * (-1.0) / determinant; // swapped with 6
260  inverted_t.values[10] = matrix_minors3x3[8] / determinant;
261  // Second, calculate inverted sub matrix T (the translation part) (-inv(R) * T)
262  // [ 12 ]
263  // -inv(R) * [ 13 ]
264  // [ 14 ]
265  // matrix cell = column(val1) * row_val1 + ... +column(val3) * row_val3
266  inverted_t.values[12] = (-inverted_t.values[0]) * m_transformation.values[12] +
267  (-inverted_t.values[4]) * m_transformation.values[13] +
268  (-inverted_t.values[8]) * m_transformation.values[14];
269  inverted_t.values[13] = (-inverted_t.values[1]) * m_transformation.values[12] +
270  (-inverted_t.values[5]) * m_transformation.values[13] +
271  (-inverted_t.values[9]) * m_transformation.values[14];
272  inverted_t.values[14] = (-inverted_t.values[2]) * m_transformation.values[12] +
273  (-inverted_t.values[6]) * m_transformation.values[13] +
274  (-inverted_t.values[10]) * m_transformation.values[14];
275  // Fill in the fixed values:
276  inverted_t.values[3] = 0;
277  inverted_t.values[7] = 0;
278  inverted_t.values[11] = 0;
279  inverted_t.values[15] = 1.0;
280  return Transformation(inverted_t);
281  */
282  const double& i00 = m_transformation.values[0];
283  const double& i01 = m_transformation.values[1];
284  const double& i02 = m_transformation.values[2];
285  //const double& i03 = m_transformation.values[3];
286  const double& i10 = m_transformation.values[4];
287  const double& i11 = m_transformation.values[5];
288  const double& i12 = m_transformation.values[6];
289  //const double& i13 = m_transformation.values[7];
290  const double& i20 = m_transformation.values[8];
291  const double& i21 = m_transformation.values[9];
292  const double& i22 = m_transformation.values[10];
293  //const double& i23 = m_transformation.values[11];
294  const double& i30 = m_transformation.values[12];
295  const double& i31 = m_transformation.values[13];
296  const double& i32 = m_transformation.values[14];
297  //const double& i33 = m_transformation.values[15];
298 
299  double s0 = i00 * i11 - i10 * i01;
300  double s1 = i00 * i12 - i10 * i02;
301  //double s2 = 0.0;
302  double s3 = i01 * i12 - i11 * i02;
303  //double s4 = 0.0;
304  //double s5 = 0.0;
305 
306  //double c5 = i22;
307  //double c4 = i21;
308  double c3 = i21 * i32 - i31 * i22;
309  //double c2 = i20;
310  double c1 = i20 * i32 - i30 * i22;
311  double c0 = i20 * i31 - i30 * i21;
312 
313  double det = s0 * i22 - s1 * i21 + s3 * i20;
314  assert(det != 0.0);
315  double invdet = 1 / det;
316 
317  inverted_t.values[0] = (i11 * i22 - i12 * i21) * invdet;
318  inverted_t.values[1] = (-i01 * i22 + i02 * i21) * invdet;
319  inverted_t.values[2] = s3 * invdet;

333  inverted_t.values[13] = (i00 * c3 - i01 * c1 + i02 * c0) * invdet;
334  inverted_t.values[14] = (-i30 * s3 + i31 * s1 - i32 * s0) * invdet;
335  inverted_t.values[15] = 1.0;
336 
337  return Transformation(inverted_t);

348 Transformation& Transformation::normalize() {
349  if (m_transformation.values[15] == 1.0) {
350  // No normalisation necessary (is already normal)
351  return (*this);
352  }
353  assert(m_transformation.values[15] != 0.0);
354  for (size_t i=0; i < 15; ++i) {
355  m_transformation.values[i] = m_transformation.values[i] / m_transformation.values[15];
356  }
357  m_transformation.values[15] = 1.0;
358  return (*this);
359 }

381 Transformation Transformation::operator*(Transformation transform) {
382  double new_matrix[4][4];
383  for (size_t col=0; col < 4; ++col) {
384  for (size_t row=0; row < 4; ++row) {
385  new_matrix[col][row] = 0.0;
386  for (size_t i=0; i < 4; ++i) {
387  new_matrix[col][row] += ((*this)[(i*4)+row] * transform[(col*4)+i]);
388  }
389  }
390  }
391  SUTransformation array_vals = {{new_matrix[0][0], new_matrix[0][1], new_matrix[0][2], new_matrix[0][3],
392  new_matrix[1][0], new_matrix[1][1], new_matrix[1][2], new_matrix[1][3],
393  new_matrix[2][0], new_matrix[2][1], new_matrix[2][2], new_matrix[2][3],
394  new_matrix[3][0], new_matrix[3][1], new_matrix[3][2], new_matrix[3][3]}};
395  return Transformation(array_vals);
396 }

514 Transformation Transformation::transformation_rotate_about_line(const double angle, const Line3D line){
515  if (!line) {
516  throw std::invalid_argument("CW::Transformation::transformation_rotate_about_line(): Line3D given is null");
517  }
518  // Solution derived from this article: http://inside.mines.edu/fs_home/gmurray/ArbitraryAxisRotation/
519  SUTransformation transform;
520  double u = line.direction.x;
521  double v = line.direction.y;
522  double w = line.direction.z;
523  double a = line.point.x;
524  double b = line.point.y;
525  double c = line.point.z;
526  double cos_angle = cos(angle);
527  double sin_angle = sin(angle);
528  double u2 = pow(u,2);
529  double v2 = pow(v,2);
530  double w2 = pow(w,2);
531  double au = a * u;
532  double av = a * v;
533  double aw = a * w;
534  double bu = b * u;
535  double bv = b * v;
536  double bw = b * w;
537  double cu = c * u;
538  double cv = c * v;
539  double cw = c * w;
540 
541  transform.values[0] = u2 + (cos_angle * (v2 + w2));
542  transform.values[1] = (u * v * (1 - cos_angle)) + (w * sin_angle);
543  transform.values[2] = (u * w * (1 - cos_angle)) - (v * sin_angle);
544  transform.values[3] = 0;
545 
546  transform.values[4] = (u * v * (1 - cos_angle)) - (w * sin_angle);
547  transform.values[5] = v2 + ((u2 + w2) * cos_angle);
548  transform.values[6] = (v * w * (1 - cos_angle)) + (u * sin_angle);
549  transform.values[7] = 0;
550 
551  transform.values[8] = (u * w * (1 - cos_angle)) + (v * sin_angle);
552  transform.values[9] = (v * w * (1 - cos_angle)) - (u * sin_angle);
553  transform.values[10] = w2 + ((u2 + v2) * cos_angle);
554  transform.values[11] = 0;
555 
556  transform.values[12] = (( (a * (v2 + w2)) - (u * (bv + cw)) ) * (1 - cos_angle)) + ((bw - cv) * sin_angle);
557  transform.values[13] = (( (b * (u2 + w2)) - (v * (au + cw)) ) * (1 - cos_angle)) + ((cu - aw) * sin_angle);
558  transform.values[14] = (( (c * (u2 + v2)) - (w * (au + bv)) ) * (1 - cos_angle)) + ((av - bu) * sin_angle);
559  transform.values[15] = 1;
560  return Transformation(transform);
561 }
